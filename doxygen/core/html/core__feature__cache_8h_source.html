224 __STATIC_FORCEINLINE int32_t GetICacheInfo(CacheInfo_Type *info)
225 {
226  if (info == NULL) {
227  return -1;
228  }
229  CSR_MICFGINFO_Type csr_ccfg = (CSR_MICFGINFO_Type)__RV_CSR_READ(CSR_MICFG_INFO);
230  info->setperway = (1 << csr_ccfg.b.set) << 3;
231  info->ways = (1 + csr_ccfg.b.way);
232  if (csr_ccfg.b.lsize == 0) {
233  info->linesize = 0;
234  } else {
235  info->linesize = (1 << (csr_ccfg.b.lsize - 1)) << 3;
236  }
237  info->size = info->setperway * info->ways * info->linesize;
238  return 0;
239 }
