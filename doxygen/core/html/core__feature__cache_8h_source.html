204 __STATIC_FORCEINLINE int32_t GetICacheInfo(CacheInfo_Type *info)
205 {
206  if (info == NULL) {
207  return -1;
208  }
209  CSR_MICFGINFO_Type csr_ccfg = (CSR_MICFGINFO_Type)__RV_CSR_READ(CSR_MICFG_INFO);
210  info->setperway = (1 << csr_ccfg.b.set) << 3;
211  info->ways = (1 + csr_ccfg.b.way);
212  if (csr_ccfg.b.lsize == 0) {
213  info->linesize = 0;
214  } else {
215  info->linesize = (1 << (csr_ccfg.b.lsize - 1)) << 3;
216  }
217  info->size = info->setperway * info->ways * info->linesize;
218  return 0;
219 }
220

237  info->setperway = (1 << csr_ccfg.b.set) << 3;

243  }
244  info->size = info->setperway * info->ways * info->linesize;
245  return 0;
246 }
