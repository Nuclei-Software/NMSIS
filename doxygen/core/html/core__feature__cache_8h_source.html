207 __STATIC_FORCEINLINE int32_t GetICacheInfo(CacheInfo_Type *info)
208 {
209  if (info == NULL) {
210  return -1;
211  }
212  CSR_MICFGINFO_Type csr_ccfg = (CSR_MICFGINFO_Type)__RV_CSR_READ(CSR_MICFG_INFO);
213  info->setperway = (1 << csr_ccfg.b.set) << 3;
214  info->ways = (1 + csr_ccfg.b.way);
215  if (csr_ccfg.b.lsize == 0) {
216  info->linesize = 0;
217  } else {
218  info->linesize = (1 << (csr_ccfg.b.lsize - 1)) << 3;
219  }
220  info->size = info->setperway * info->ways * info->linesize;
221  return 0;
222 }
