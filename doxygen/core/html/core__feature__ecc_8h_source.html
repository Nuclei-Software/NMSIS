956 static uint8_t ECC_GenerateECCCodeU32(uint32_t a)
957 {
958  uint8_t ecc_bits = 0;
959  uint8_t ecc_bit_0 = ((a >> 0) & 1) ^ ((a >> 1) & 1) ^ ((a >> 3) & 1) ^ ((a >> 4) & 1) ^
960  ((a >> 6) & 1) ^ ((a >> 8) & 1) ^ ((a >> 10) & 1) ^ ((a >> 12) & 1) ^
961  ((a >> 14) & 1) ^ ((a >> 17) & 1) ^ ((a >> 19) & 1) ^ ((a >> 20) & 1) ^
962  ((a >> 24) & 1) ^ ((a >> 28) & 1);
