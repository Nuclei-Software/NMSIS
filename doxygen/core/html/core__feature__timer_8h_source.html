117 __STATIC_FORCEINLINE void SysTimer_SetLoadValue(uint64_t value)
118 {
119 #if __RISCV_XLEN == 32
120  uint8_t *addr;
121  addr = (uint8_t *)(&(SysTimer->MTIMER));
122  __SW(addr, 0); // prevent carry
123  __SW(addr + 4, (uint32_t)(value >> 32));
124  __SW(addr, (uint32_t)(value));
125 #else
126  SysTimer->MTIMER = value;
127 #endif
128 }

139 __STATIC_FORCEINLINE uint64_t SysTimer_GetLoadValue(void)
140 {
141 #if __RISCV_XLEN == 32
142  volatile uint32_t high0, low, high;
143  uint64_t full;
144  uint8_t *addr;
145 
146  addr = (uint8_t *)(&(SysTimer->MTIMER));
147 
148  high0 = __LW(addr + 4);
149  low = __LW(addr);
150  high = __LW(addr + 4);
151  if (high0 != high) {
152  low = __LW(addr);
153  }
154  full = (((uint64_t)high) << 32) | low;
155  return full;
156 #else
157  return SysTimer->MTIMER;
158 #endif
159 }

174 __STATIC_FORCEINLINE void SysTimer_SetHartCompareValue(uint64_t value, unsigned long hartid)
175 {
176  if (hartid == 0) {
177 #if __RISCV_XLEN == 32
178  uint8_t *addr;
179  addr = (uint8_t *)(&(SysTimer->MTIMERCMP));
180  __SW(addr, -1U); // prevent load > timecmp
181  __SW(addr + 4, (uint32_t)(value >> 32));
182  __SW(addr, (uint32_t)(value));
183 #else
184  SysTimer->MTIMERCMP = value;
185 #endif
186  } else {
187  uint8_t *addr = (uint8_t *)(SysTimer_CLINT_MTIMECMP_BASE(hartid));
188 #if __RISCV_XLEN == 32
189  __SW(addr, -1U); // prevent load > timecmp
190  __SW(addr + 4, (uint32_t)(value >> 32));
191  __SW(addr, (uint32_t)value);
192 #else
193  __SD(addr, value);
194 #endif
195  }
196 }

599 __STATIC_FORCEINLINE uint32_t SysTick_Reload(uint64_t ticks)
600 {
601  uint64_t cur_ticks = SysTimer_GetLoadValue();
602  uint64_t reload_ticks = ticks + cur_ticks;
603 
604  if (__USUALLY(reload_ticks > cur_ticks)) {
605  SysTimer_SetCompareValue(reload_ticks);
606  } else {
607  /* When added the ticks value, then the MTIMERCMP < TIMER,
608  * which means the MTIMERCMP is overflowed,
609  * so we need to reset the counter to zero */
610  SysTimer_SetLoadValue(0);
611  SysTimer_SetCompareValue(ticks);
612  }
613 
614  return (0UL);
615 }
