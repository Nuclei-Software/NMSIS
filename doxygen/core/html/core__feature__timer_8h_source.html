129 __STATIC_FORCEINLINE uint64_t SysTimer_GetLoadValue(void)
130 {
131 #if __RISCV_XLEN == 32
132  volatile uint32_t high0, low, high;
133  uint64_t full;
134  void *addr;
135 
136  addr = (void *)(&(SysTimer->MTIMER));
137 
138  high0 = __LW(addr + 4);
139  low = __LW(addr);
140  high = __LW(addr + 4);
141  if (high0 != high) {
142  low = __LW(addr);
143  }
144  full = (((uint64_t)high) << 32) | low;
145  return full;
146 #else
147  return SysTimer->MTIMER;
148 #endif
149 }

162 __STATIC_FORCEINLINE void SysTimer_SetCompareValue(uint64_t value)
163 {
164  unsigned long hartid = __RV_CSR_READ(CSR_MHARTID);
165  if (hartid == 0) {
166  SysTimer->MTIMERCMP = value;
167  } else {
168  void *addr = (void *)(SysTimer_CLINT_MTIMECMP_BASE(hartid));
169 #if __RISCV_XLEN == 32
170  __SW(addr, (uint32_t)value);
171  __SW(addr + 4, (uint32_t)(value >> 32));
172 #else
173  __SD(addr, value);
174 #endif
175  }
176 }
177 
187 __STATIC_FORCEINLINE uint64_t SysTimer_GetCompareValue(void)
188 {
189  unsigned long hartid = __RV_CSR_READ(CSR_MHARTID);
190  if (hartid == 0) {
191  return SysTimer->MTIMERCMP;
192  } else {
193  uint64_t full;
194  void *addr = (void *)(SysTimer_CLINT_MTIMECMP_BASE(hartid));
195 #if __RISCV_XLEN == 32
196  // MTIMECMP didn't increase
197  uint32_t high, low;
198 
199  high = __LW(addr + 4);
200  low = __LW(addr);
201  full = (((uint64_t)high) << 32) | low;
202 #else
203  full = __LD(addr);
204 #endif
205  return full;
206  }
207 }

407 __STATIC_INLINE uint32_t SysTick_Config(uint64_t ticks)
408 {
409  uint64_t loadticks = SysTimer_GetLoadValue();
410  SysTimer_SetCompareValue(ticks + loadticks);
411  ECLIC_SetShvIRQ(SysTimer_IRQn, ECLIC_NON_VECTOR_INTERRUPT);
412  ECLIC_SetLevelIRQ(SysTimer_IRQn, 0);
413  ECLIC_EnableIRQ(SysTimer_IRQn);
414  return (0UL);
415 }

440 __STATIC_FORCEINLINE uint32_t SysTick_Reload(uint64_t ticks)
441 {
442  uint64_t cur_ticks = SysTimer_GetLoadValue();
443  uint64_t reload_ticks = ticks + cur_ticks;
444 
445  if (__USUALLY(reload_ticks > cur_ticks)) {
446  SysTimer_SetCompareValue(reload_ticks);
447  } else {
448  /* When added the ticks value, then the MTIMERCMP < TIMER,
449  * which means the MTIMERCMP is overflowed,
450  * so we need to reset the counter to zero */
451  SysTimer_SetLoadValue(0);
452  SysTimer_SetCompareValue(ticks);
453  }
454 
455  return (0UL);
456 }
