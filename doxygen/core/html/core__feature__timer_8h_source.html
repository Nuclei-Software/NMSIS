64 /* Timer Control / Status Register Definitions */
65 #define SysTimer_MTIMECTL_TIMESTOP_Pos 0U
66 #define SysTimer_MTIMECTL_TIMESTOP_Msk (1UL << SysTimer_MTIMECTL_TIMESTOP_Pos)

117 __STATIC_FORCEINLINE void SysTimer_SetLoadValue(uint64_t value)
118 {
119 #if __RISCV_XLEN == 32
120  uint8_t *addr;
121  addr = (uint8_t *)(&(SysTimer->MTIMER));
122  __SW(addr, 0); // prevent carry
123  __SW(addr + 4, (uint32_t)(value >> 32));
124  __SW(addr, (uint32_t)(value));
125 #else
126  SysTimer->MTIMER = value;
127 #endif
128 }

139 __STATIC_FORCEINLINE uint64_t SysTimer_GetLoadValue(void)
140 {
141 #if __RISCV_XLEN == 32
142  volatile uint32_t high0, low, high;
143  uint64_t full;
144  uint8_t *addr;
145 
146  addr = (uint8_t *)(&(SysTimer->MTIMER));
147 
148  high0 = __LW(addr + 4);
149  low = __LW(addr);
150  high = __LW(addr + 4);
151  if (high0 != high) {
152  low = __LW(addr);
153  }
154  full = (((uint64_t)high) << 32) | low;
155  return full;
156 #else
157  return SysTimer->MTIMER;
158 #endif
159 }

172 __STATIC_FORCEINLINE void SysTimer_SetCompareValue(uint64_t value)
173 {
174  unsigned long hartid = __RV_CSR_READ(CSR_MHARTID);
175  if (hartid == 0) {
176 #if __RISCV_XLEN == 32
177  uint8_t *addr;
178  addr = (uint8_t *)(&(SysTimer->MTIMERCMP));
179  __SW(addr, -1U); // prevent load > timecmp
180  __SW(addr + 4, (uint32_t)(value >> 32));
181  __SW(addr, (uint32_t)(value));
182 #else
183  SysTimer->MTIMERCMP = value;
184 #endif
185  } else {
186  uint8_t *addr = (uint8_t *)(SysTimer_CLINT_MTIMECMP_BASE(hartid));
187 #if __RISCV_XLEN == 32
188  __SW(addr, -1U); // prevent load > timecmp
189  __SW(addr + 4, (uint32_t)(value >> 32));
190  __SW(addr, (uint32_t)value);
191 #else
192  __SD(addr, value);
193 #endif
194  }
195 }

206 __STATIC_FORCEINLINE uint64_t SysTimer_GetCompareValue(void)
207 {
208  unsigned long hartid = __RV_CSR_READ(CSR_MHARTID);
209  if (hartid == 0) {
210  return SysTimer->MTIMERCMP;
211  } else {
212  uint64_t full;
213  uint8_t *addr = (uint8_t *)(SysTimer_CLINT_MTIMECMP_BASE(hartid));
214 #if __RISCV_XLEN == 32
215  // MTIMECMP didn't increase
216  uint32_t high, low;
217 
218  high = __LW(addr + 4);
219  low = __LW(addr);
220  full = (((uint64_t)high) << 32) | low;
221 #else
222  full = __LD(addr);
223 #endif
224  return full;
225  }
226 }

426 __STATIC_INLINE uint32_t SysTick_Config(uint64_t ticks)
427 {
428  uint64_t loadticks = SysTimer_GetLoadValue();
429  SysTimer_SetCompareValue(ticks + loadticks);
430  ECLIC_SetShvIRQ(SysTimer_IRQn, ECLIC_NON_VECTOR_INTERRUPT);
431  ECLIC_SetLevelIRQ(SysTimer_IRQn, 0);
432  ECLIC_EnableIRQ(SysTimer_IRQn);
433  return (0UL);

459 __STATIC_FORCEINLINE uint32_t SysTick_Reload(uint64_t ticks)
460 {
461  uint64_t cur_ticks = SysTimer_GetLoadValue();
462  uint64_t reload_ticks = ticks + cur_ticks;
463 
464  if (__USUALLY(reload_ticks > cur_ticks)) {
465  SysTimer_SetCompareValue(reload_ticks);
466  } else {
467  /* When added the ticks value, then the MTIMERCMP < TIMER,
468  * which means the MTIMERCMP is overflowed,
469  * so we need to reset the counter to zero */
470  SysTimer_SetLoadValue(0);
471  SysTimer_SetCompareValue(ticks);
472  }
473 
474  return (0UL);
475 }
