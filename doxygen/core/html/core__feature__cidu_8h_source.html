186 __STATIC_FORCEINLINE long CIDU_SetFirstClaimMode(uint32_t int_id, uint32_t core_id)
187 {
188  uint32_t val = 0;
189  uint32_t* addr = (uint32_t*)CIDU_INT_MASK_ADDR(int_id);
190  uint32_t mask = 1 << core_id;
191 
192  __SW(addr, mask);
193  val = __LW(addr);

195  return 0;
196  }
197  return -1;
