634 __STATIC_FORCEINLINE void __ECLIC_SetLevelIRQ(IRQn_Type IRQn, uint8_t lvl_abs)
635 {
636  uint8_t nlbits = __ECLIC_GetCfgNlbits();
637  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
638 
639  if (nlbits == 0) {
640  return;
641  }
642 
643  if (nlbits > intctlbits) {
644  nlbits = intctlbits;
645  }
646  uint8_t maxlvl = ((1 << nlbits) - 1);
647  if (lvl_abs > maxlvl) {
648  lvl_abs = maxlvl;
649  }
650  uint8_t lvl = lvl_abs << (ECLIC_MAX_NLBITS - nlbits);
651  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
652  cur_ctrl = cur_ctrl << nlbits;
653  cur_ctrl = cur_ctrl >> nlbits;
654  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | lvl));
655 }

699 __STATIC_FORCEINLINE void __ECLIC_SetPriorityIRQ(IRQn_Type IRQn, uint8_t pri)
700 {
701  uint8_t nlbits = __ECLIC_GetCfgNlbits();
702  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
703  if (nlbits < intctlbits) {
704  uint8_t maxpri = ((1 << (intctlbits - nlbits)) - 1);
705  if (pri > maxpri) {
706  pri = maxpri;
707  }
708  pri = pri << (ECLIC_MAX_NLBITS - intctlbits);
709  uint8_t mask = ((uint8_t)(-1)) >> intctlbits;
710  pri = pri | mask;
711  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
712  cur_ctrl = cur_ctrl >> (ECLIC_MAX_NLBITS - nlbits);
713  cur_ctrl = cur_ctrl << (ECLIC_MAX_NLBITS - nlbits);
714  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | pri));
715  }
716 }

730 {
731  uint8_t nlbits = __ECLIC_GetCfgNlbits();

734  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
