581 __STATIC_FORCEINLINE void __ECLIC_SetLevelIRQ(IRQn_Type IRQn, uint8_t lvl_abs)
582 {
583  uint8_t nlbits = __ECLIC_GetCfgNlbits();
584  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
585 
586  if (nlbits == 0) {
587  return;
588  }
589 
590  if (nlbits > intctlbits) {
591  nlbits = intctlbits;
592  }
593  uint8_t maxlvl = ((1 << nlbits) - 1);
594  if (lvl_abs > maxlvl) {
595  lvl_abs = maxlvl;
596  }
597  uint8_t lvl = lvl_abs << (ECLIC_MAX_NLBITS - nlbits);
598  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
599  cur_ctrl = cur_ctrl << nlbits;
600  cur_ctrl = cur_ctrl >> nlbits;
601  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | lvl));
602 }

646 __STATIC_FORCEINLINE void __ECLIC_SetPriorityIRQ(IRQn_Type IRQn, uint8_t pri)
647 {
648  uint8_t nlbits = __ECLIC_GetCfgNlbits();
649  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
650  if (nlbits < intctlbits) {
651  uint8_t maxpri = ((1 << (intctlbits - nlbits)) - 1);
652  if (pri > maxpri) {
653  pri = maxpri;
654  }
655  pri = pri << (ECLIC_MAX_NLBITS - intctlbits);
656  uint8_t mask = ((uint8_t)(-1)) >> intctlbits;
657  pri = pri | mask;
658  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
659  cur_ctrl = cur_ctrl >> (ECLIC_MAX_NLBITS - nlbits);
660  cur_ctrl = cur_ctrl << (ECLIC_MAX_NLBITS - nlbits);
661  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | pri));
662  }
663 }
