584 __STATIC_FORCEINLINE void __ECLIC_SetLevelIRQ(IRQn_Type IRQn, uint8_t lvl_abs)
585 {
586  uint8_t nlbits = __ECLIC_GetCfgNlbits();
587  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
588 
589  if (nlbits == 0) {
590  return;
591  }
592 
593  if (nlbits > intctlbits) {
594  nlbits = intctlbits;
595  }
596  uint8_t maxlvl = ((1 << nlbits) - 1);
597  if (lvl_abs > maxlvl) {
598  lvl_abs = maxlvl;
599  }
600  uint8_t lvl = lvl_abs << (ECLIC_MAX_NLBITS - nlbits);
601  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
602  cur_ctrl = cur_ctrl << nlbits;
603  cur_ctrl = cur_ctrl >> nlbits;
604  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | lvl));
605 }

649 __STATIC_FORCEINLINE void __ECLIC_SetPriorityIRQ(IRQn_Type IRQn, uint8_t pri)
650 {
651  uint8_t nlbits = __ECLIC_GetCfgNlbits();
652  uint8_t intctlbits = (uint8_t)__ECLIC_INTCTLBITS;
653  if (nlbits < intctlbits) {
654  uint8_t maxpri = ((1 << (intctlbits - nlbits)) - 1);
655  if (pri > maxpri) {
656  pri = maxpri;
657  }
658  pri = pri << (ECLIC_MAX_NLBITS - intctlbits);
659  uint8_t mask = ((uint8_t)(-1)) >> intctlbits;
660  pri = pri | mask;
661  uint8_t cur_ctrl = __ECLIC_GetCtrlIRQ(IRQn);
662  cur_ctrl = cur_ctrl >> (ECLIC_MAX_NLBITS - nlbits);
663  cur_ctrl = cur_ctrl << (ECLIC_MAX_NLBITS - nlbits);
664  __ECLIC_SetCtrlIRQ(IRQn, (cur_ctrl | pri));
665  }
