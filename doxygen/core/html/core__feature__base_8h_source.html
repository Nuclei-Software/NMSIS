588 __STATIC_FORCEINLINE void __switch_mode(uint8_t mode, uintptr_t stack, void(*entry_point)(void))
589 {
590  unsigned long val = 0;
591 
592  /* Set MPP to the requested privilege mode */
593  val = __RV_CSR_READ(CSR_MSTATUS);
594  val = __RV_INSERT_FIELD(val, MSTATUS_MPP, mode);
595 
596  /* Set previous MIE disabled */
597  val = __RV_INSERT_FIELD(val, MSTATUS_MPIE, 0);
598 
599  __RV_CSR_WRITE(CSR_MSTATUS, val);
600 
601  /* Set the entry point in MEPC */
602  __RV_CSR_WRITE(CSR_MEPC, entry_point);
603 
604  /* Set the register file */
605  __ASM volatile("mv sp, %0" ::"r"(stack));
606 
607  __ASM volatile("mret");
608 }
