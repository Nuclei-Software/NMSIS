130 __STATIC_INLINE uint8_t __get_sPMPxCFG(uint32_t entry_idx)
131 {
132  rv_csr_t spmpcfgx = 0;
133  uint8_t csr_cfg_num = 0;
134  uint16_t csr_idx = 0;
135  uint16_t cfg_shift = 0;
136 
137  if (entry_idx >= __SPMP_ENTRY_NUM) return 0;
138 
139 #if __RISCV_XLEN == 32
140  csr_cfg_num = 4;
141  csr_idx = entry_idx >> 2;
142 #elif __RISCV_XLEN == 64
143  csr_cfg_num = 8;
144  /* For RV64, spmpcfg0 and spmpcfg2 each hold 8 sPMP entries, align by 2 */
145  csr_idx = (entry_idx >> 2) & ~1;
146 #else
147  // TODO Add RV128 Handling
148  return 0;
149 #endif
150  spmpcfgx = __get_sPMPCFGx(csr_idx);
151  /*
152  * first get specific spmpxcfg's order in one CSR composed of csr_cfg_num spmpxcfgs,
153  * then get spmpxcfg's bit position in one CSR by left shift 3(each spmpxcfg size is one byte)
154  */
155  cfg_shift = (entry_idx & (csr_cfg_num - 1)) << 3;
156 
157  /* read specific spmpxcfg register value */
158  return (uint8_t)(__RV_EXTRACT_FIELD(spmpcfgx, 0xFF << cfg_shift));
159 }

270 __STATIC_INLINE void __set_sPMPENTRYx(uint32_t entry_idx, const spmp_config *spmp_cfg)
271 {
272  unsigned int cfg_shift, cfg_csr_idx, addr_csr_idx = 0;
273  unsigned long cfgmask, addrmask = 0;
274  unsigned long spmpcfg, spmpaddr = 0;
275  uint8_t protection, csr_cfg_num = 0;
276  /* check parameters */
277  if (entry_idx >= __SPMP_ENTRY_NUM || spmp_cfg->order > __RISCV_XLEN || spmp_cfg->order < SPMP_SHIFT) return;
278 
279  /* calculate sPMP register and offset */
280 #if __RISCV_XLEN == 32
281  csr_cfg_num = 4;
282  cfg_csr_idx = (entry_idx >> 2);
283 #elif __RISCV_XLEN == 64
284  csr_cfg_num = 8;
285  cfg_csr_idx = ((entry_idx >> 2)) & ~1;
286 #else
287  // TODO Add RV128 Handling
288  return;
289 #endif
290  /*
291  * first get specific spmpxcfg's order in one CSR composed of csr_cfg_num spmpxcfgs,
292  * then get spmpxcfg's bit position in one CSR by left shift 3, each spmpxcfg size is one byte
293  */
294  cfg_shift = (entry_idx & (csr_cfg_num - 1)) << 3;
295  addr_csr_idx = entry_idx;
296 
297  /* encode sPMP config */
298  protection = spmp_cfg->protection;
299  protection |= (SPMP_SHIFT == spmp_cfg->order) ? SPMP_A_NA4 : SPMP_A_NAPOT;
300  cfgmask = ~(0xFFUL << cfg_shift);
301  spmpcfg = (__get_sPMPCFGx(cfg_csr_idx) & cfgmask);
302  spmpcfg |= ((protection << cfg_shift) & ~cfgmask);
303 
304  /* encode sPMP address */
305  if (SPMP_SHIFT == spmp_cfg->order) { /* NA4 */
306  spmpaddr = (spmp_cfg->base_addr >> SPMP_SHIFT);
307  } else { /* NAPOT */
308  addrmask = (1UL << (spmp_cfg->order - SPMP_SHIFT)) - 1;
309  spmpaddr = ((spmp_cfg->base_addr >> SPMP_SHIFT) & ~addrmask);
310  spmpaddr |= (addrmask >> 1);
311  }
312  /*
313  * write csrs, update the address first, in case the entry is locked that
314  * we won't be able to modify it after we set the config csr.
315  */
316  __set_sPMPADDRx(addr_csr_idx, spmpaddr);
317  __set_sPMPCFGx(cfg_csr_idx, spmpcfg);
318 }

331 __STATIC_INLINE int __get_sPMPENTRYx(unsigned int entry_idx, spmp_config *spmp_cfg)
332 {
333  unsigned int cfg_shift, cfg_csr_idx, addr_csr_idx = 0;
334  unsigned long cfgmask, spmpcfg, prot = 0;
335  unsigned long t1, addr, spmpaddr, len = 0;
336  uint8_t csr_cfg_num = 0;
337  /* check parameters */
338  if (entry_idx >= __SPMP_ENTRY_NUM || !spmp_cfg) return -1;
339 
340  /* calculate sPMP register and offset */
341 #if __RISCV_XLEN == 32
342  csr_cfg_num = 4;
343  cfg_csr_idx = entry_idx >> 2;
344 #elif __RISCV_XLEN == 64
345  csr_cfg_num = 8;
346  cfg_csr_idx = (entry_idx >> 2) & ~1;
347 #else
348 // TODO Add RV128 Handling
349  return -1;
350 #endif
351 
352  cfg_shift = (entry_idx & (csr_cfg_num - 1)) << 3;
353  addr_csr_idx = entry_idx;
354 
355  /* decode sPMP config */
356  cfgmask = (0xFFUL << cfg_shift);
357  spmpcfg = (__get_sPMPCFGx(cfg_csr_idx) & cfgmask);
358  prot = spmpcfg >> cfg_shift;
359 
360  /* decode sPMP address */
361  spmpaddr = __get_sPMPADDRx(addr_csr_idx);
362  if (SPMP_A_NAPOT == (prot & SPMP_A)) {
363  t1 = __CTZ(~spmpaddr);
364  addr = (spmpaddr & ~((1UL << t1) - 1)) << SPMP_SHIFT;
365  len = (t1 + SPMP_SHIFT + 1);
366  } else {
367  addr = spmpaddr << SPMP_SHIFT;
368  len = SPMP_SHIFT;
369  }
370 
371  /* return details */
372  spmp_cfg->protection = prot;
373  spmp_cfg->base_addr = addr;
374  spmp_cfg->order = len;
375 
376  return 0;
377 }
